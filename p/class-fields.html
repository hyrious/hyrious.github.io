Class Computed Fields

Last updated at 2021-08-15

For typescript: see useDefineForClassFields.

var foo = 'foo'

class A {
  [foo] = bar      // [foo] === 'foo', bar is calculated at `new ()`
}

foo = 'bar'
var bar = 1

console.log(new A) // A { foo: 1 }

class B {
  [foo]            // equals to: [foo] = undefined
}

console.log(new B) // B { bar: undefined }
